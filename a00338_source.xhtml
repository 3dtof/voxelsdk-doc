84  void run()
85  {
86  //logger(LOG_INFO) << "USBBulkStreamer: Running Capture" << std::endl;
87  if (!usbIO->resetBulkEndPoint(endpoint))
88  {
89  logger(LOG_ERROR) << "USBBulkStreamer: Bulk End point reset failed" << std::endl;
90  return;
91  }
92 
93  while(captureRunning)
94  {
95  if(usbIO->bulkTransfer(endpoint, usbBuffer.data(), bufSize + BULK_XFER_EXTRA_SIZE, transferred, timeout))
96  {
97  if(transferred == bufSize)
98  {
99  validFrames++;
100  retriesCount = 0;
101  putBuffer();
102  }
103  else
104  {
105  droppedFrames++;
106  if(droppedFrames % 100 == 0)
107  logger(LOG_ERROR) << "USBBulkStreamer: Dropped frames " << droppedFrames
108  << " Valid frames " << validFrames << std::endl;
109  }
110  }
111  else
112  {
113  retriesCount++;
114  if(retriesCount == 3)
115  {
116  if (!usbIO->resetBulkEndPoint(endpoint))
117  {
118  logger(LOG_ERROR) << "USBBulkStreamer: Bulk End point reset failed" << std::endl;
119  captureRunning = false;
120  }
121 
122  logger(LOG_WARNING) << "USBBulkStreamer: Resetting bulk endpoint at " << _timer.getCurentRealTime() << std::endl;
123  retriesCount = 0;
124  resetCount++;
125  if(resetCount % 100 == 0)
126  {
127  logger(LOG_ERROR) << "USBBulkStreamer: Did not get a frame in 100 attempts" << std::endl;
128  captureRunning = false;
129  }
130  }
