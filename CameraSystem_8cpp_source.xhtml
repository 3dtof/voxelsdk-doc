43 void CameraSystem::_loadLibraries(const Vector<String> &paths)
44 {
45  Vector<String> files;
46 
47  int numberOfLoadLibraries = 0;
48 
49  for(auto &path: paths)
50  {
51  getFiles(path, LIB_MATCH_STRING, files);
52 
53  for(auto &file: files)
54  {
55  std::ifstream ff(file, std::ios::binary);
56 
57  if(ff.good()) // File readable
58  {
59  ff.close();
60 
61  DepthCameraLibraryPtr p(new DepthCameraLibrary(file));
62 
63  if(!p->load())
64  {
65  logger(LOG_WARNING) << "CameraSystem: Failed to load library " << file << ". Ignoring it." << std::endl;
66  continue;
67  }
68 
69  if (p->getABIVersion() != VOXEL_ABI_VERSION)
70  {
71  logger(LOG_WARNING) << "CameraSystem: Ignoring Voxel library " << file << " with ABI version = " << p->getABIVersion() << ". Expected ABI version = " << VOXEL_ABI_VERSION << std::endl;
72  continue;
73  }
74 
75  bool success = false;
76 
77  DepthCameraFactoryPtr factory = p->getDepthCameraFactory();
78 
79  if(!factory || !addDepthCameraFactory(factory))
80  logger(LOG_WARNING) << "CameraSystem: Failed to load or register a depth camera factory from library " << file <<". Ignoring this library." << std::endl;
81  else
82  success = true;
83 
84  FilterFactoryPtr filterFactory = p->getFilterFactory();
85 
86  if(!filterFactory || !addFilterFactory(filterFactory))
87  logger(LOG_DEBUG) << "CameraSystem: Could not find filter factory" << std::endl;
88  else
89  success = true;
90 
91  DownloaderFactoryPtr downloaderFactory = p->getDownloaderFactory();
92 
93  if(!downloaderFactory || !addDownloaderFactory(downloaderFactory))
94  logger(LOG_DEBUG) << "CameraSystem: Could not find downloader factory" << std::endl;
95  else
96  success = true;
97 
98  if(success)
99  {
100  // Preserving this library
101  _libraries.push_back(p);
102  numberOfLoadLibraries++;
103  logger(LOG_INFO) << "CameraSystem: Successfully loaded factory from library " << file << std::endl;
104  }
105  }
106  else
107  logger(LOG_WARNING) << "CameraSystem: Library file " << file << " is not readable. Ignoring it." << std::endl;
108  }
109  }
110 
111  if(!numberOfLoadLibraries)
112  logger(LOG_WARNING) << "CameraSystem: No depth camera library found or loaded." << std::endl;
113 }

116 bool CameraSystem::addDepthCameraFactory(DepthCameraFactoryPtr factory)
117 {
118  // Adding factory to the "_factories" map for later query
119  const Vector<DevicePtr> &devices = factory->getSupportedDevices();
120 
121  if(!devices.size()) // No devices?
122  return false;
123 
124  int numberOfDevicesAdded = 0;
125 
126  for(auto &device: devices)
127  {
128  if(device->serialNumber().size())
129  {
130  logger(LOG_WARNING) << "CameraSystem: Device type " << device->id() << " being registered from factory has serial number. Ignoring it." << std::endl;
131  continue;
132  }
133 
134  auto f = _factories.find(device->id());
135  if(f != _factories.end())
136  {
137  logger(LOG_WARNING) << "CameraSystem: Device type " << device->id() << " already has a factory '" << f->second->name() << "'. Not overwriting it." << std::endl;
138  continue;
139  }
140 
141  numberOfDevicesAdded++;
142 
143  _factories[device->id()] = factory;
144  }
145 
146  if(!numberOfDevicesAdded)
147  {
148  logger(LOG_WARNING) << "CameraSystem: No devices added from factory '" << factory->name() << "'. Ignoring this factory." << std::endl;
149  return false;
150  }
151 
152  Vector<GeneratorIDType> supportedGeneratorIDs = factory->getSupportedGeneratorTypes();
153 
154  for(auto &d: supportedGeneratorIDs)
155  {
156  auto f = _factoryForGeneratorID.find(d);
157 
158  if(f != _factoryForGeneratorID.end())
159  {
160  logger(LOG_DEBUG) << "CameraSystem: A factory is already registered for generator ID '" << d << "'. Not replacing it." << std::endl;
161  continue;
162  }
163 
164  _factoryForGeneratorID[d] = factory;
165  }
166 
167  return true;
168 }
