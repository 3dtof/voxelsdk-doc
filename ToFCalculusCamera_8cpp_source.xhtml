310 bool ToFCalculusCamera::_updateEasyConf()
311 {
312  uint intg_duty_cycle, pix_cnt_max, sub_frame_cnt_max;
313 
314  // This represents the original value of intg_phase_pix2, and we pull in the register value by 50 cycles
315  // other values are pushed out relative to this original value
316  uint tm_intg_phase_pix2_int, tm_intg_phase_pix2_limit;
317  bool tg_en_value;
318  uint total_length_of_rdout_frame = 4950U;
319  bool pix_cnt_overflow = false, intg_phase_overflow = false;
320 
321  _get(INTEGRATION_DUTY_CYCLE, intg_duty_cycle);
322  _get(TM_PIX_CNT_MAX, pix_cnt_max);
323  _get(TM_UFRAME_CNT_MAX, sub_frame_cnt_max);
324  _get(TG_EN, tg_en_value);
325 
326  tm_intg_phase_pix2_int = 722 + ((pix_cnt_max * intg_duty_cycle) >> 6);
327 
328  if (pix_cnt_max < (total_length_of_rdout_frame + 922U))
329  pix_cnt_overflow = true;
330 
331  if (pix_cnt_overflow)
332  tm_intg_phase_pix2_limit = 723U;
333  else
334  tm_intg_phase_pix2_limit = pix_cnt_max - (total_length_of_rdout_frame + 300);
335 
336  if (tm_intg_phase_pix2_limit < tm_intg_phase_pix2_int)
337  intg_phase_overflow = true;
338  else
339  intg_phase_overflow = false;
340 
341  if (intg_phase_overflow) {
342  tm_intg_phase_pix2_int = tm_intg_phase_pix2_limit;
343  _set(INTG_DUTY_CYCLE_SET_FAILED, true);
344  return true;
345  }
346 
347  _set(INTG_DUTY_CYCLE_SET_FAILED, false);
348  _set(TG_EN, false);
349  // Set INTG_PHASE_PIX2 and RDOUT_PHASE_PIX2 to 50 cycles earlier than usual, and gate shutter at the usual point
350  _programmer->writeRegister(TM_INTG_PHASE_PIX2_REG, tm_intg_phase_pix2_int - 50);
351  _programmer->writeRegister(TM_RDOUT_PHASE_PIX2_REG, tm_intg_phase_pix2_int - 50);
352  _programmer->writeRegister(TM_SPARE3_PIX2_REG, tm_intg_phase_pix2_int);
353  _programmer->writeRegister(TM_RDOUT_HD_CNT_START_PIX1_REG, tm_intg_phase_pix2_int + 100);
354  _programmer->writeRegister(TM_RDOUT_VD_PIX1_REG, 0x400000 + tm_intg_phase_pix2_int + 100);
355  _programmer->writeRegister(TM_RDOUT_VD_PIX2_REG, tm_intg_phase_pix2_int + 100 + 8);
356  _programmer->writeRegister(TM_RDOUT_HD_CNT_END_PIX1_REG, tm_intg_phase_pix2_int + 100 + 4950);
357 
358  _programmer->writeRegister(TM_PIX_CNT_MAX_REG, pix_cnt_max);
359  _programmer->writeRegister(TM_UFRAME_CNT_MAX_REG, 4 + (sub_frame_cnt_max << 4));
360  _set(TG_EN, tg_en_value);
361 
362  return true;
363 }

384  return _parameterInit = false;

388  {

392  p->setAddress((0x5C << 8) + (p->address() & 0xFF));
393  }
394 
395  if(!_addParameters(params))

404  ParameterPtr(new CalculusIntegDutyCycleSetFailed(*this, *_programmer)),

406  return false;
407 
408  if(!ToFCamera::_init())
409  {

413  return true;
