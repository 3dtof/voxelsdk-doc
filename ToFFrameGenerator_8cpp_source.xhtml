357 bool ToFFrameGenerator::_applyPhaseOffsetCorrection(Vector<uint16_t> &phaseData)
358 {
359  if(!_phaseOffsetCorrectionData.size())
360  return true; // Nothing to do
361 
362  int16_t *_phaseOffsetCorrection;
363 
364  if(_phaseOffsetCorrectionData.size() == _maxFrameSize.height*_maxFrameSize.width)
365  _phaseOffsetCorrection = _phaseOffsetCorrectionData.data();
366  else if(_phaseOffsetCorrectionData.size() == _maxFrameSize.height*_maxFrameSize.width + 2) // contains rows and columns information?
367  _phaseOffsetCorrection = _phaseOffsetCorrectionData.data() + 2;
368  else
369  return false;
370 
371  if(phaseData.size() != (_size.width)*(_size.height))
372  return false;
373 
374  int i = _size.height, j;
375 
376  int16_t v;
377  uint16_t *d = phaseData.data();
378  int16_t *o = _phaseOffsetCorrection + _roi.x + _roi.y*_maxFrameSize.width, *o1;
379  o1 = o;
380 
381 
382  while(i--)
383  {
384  j = _size.width;
385  while(j--)
386  {
387  v = *d - *o;
388  if(v < 0)
389  *d = 0;
390  else if(v >= MAX_PHASE_VALUE + 1)
391  *d = MAX_PHASE_VALUE;
392  else
393  *d = v;
394  d++;
395  o += _columnsToMerge;
396  }
397  o1 += _rowsToMerge*_maxFrameSize.width;
398  o = o1;
399  }
400 
401  return true;

609 bool ToFFrameGenerator::generate(const ToFRawIQFramePtr &in, FramePtr &out)
610 {

612 
613  if(!input)
614  {
615  logger(LOG_ERROR) << "ToFFrameGenerator: Input data frame is not of raw ToF IQ type." << std::endl;
616  return false;
617  }
618 
619  if(input->size != _size)
620  {
621  logger(LOG_ERROR) << "ToFFrameGenerator: Input data size differs from expected size. Input size = "
622  << input->size.width << "x" << input->size.height
623  << ", expected size = " << _size.width << "x" << _size.height
624  << std::endl;
625  return false;
626  }
627 
628  ToFRawFrameTemplate<uint16_t, uint8_t> *t;
629  if(!_crossTalkFilter)
630  {
631  t = dynamic_cast<ToFRawFrameTemplate<uint16_t, uint8_t> *>(out.get());
632 
633  if(!t)
634  {
635  t = new ToFRawFrameTemplate<uint16_t, uint8_t>();
636  out = FramePtr(t);
637  }
638  }
639  else

641  t = dynamic_cast<ToFRawFrameTemplate<uint16_t, uint8_t> *>(_filterInputFrame.get());
642 
643  if(!t)
644  {
645  t = new ToFRawFrameTemplate<uint16_t, uint8_t>();
646  _filterInputFrame = FramePtr(t);
647  }
648  }
649 
650  t->size = _size;
651  t->id = input->id;
652  t->timestamp = input->timestamp;
653 
654  t->_ambient.resize(_size.width*_size.height);
655  t->_amplitude.resize(_size.width*_size.height);
656  t->_phase.resize(_size.width*_size.height);
657  t->_flags.resize(_size.width*_size.height);
658 
659  auto index = 0;
660 
661  Complex c;
662  float phase;
663 
664  for (auto l = 0; l < _size.height; l++)
665  {
666  for (auto j = 0; j < _size.width; j++)
667  {
668  index = l*_size.width + j;
669 
670  c.real(input->_i[index]);
671  c.imag(input->_q[index]);
672 
673  phase = std::arg(c);
674 
675  t->_phase[index] = ((phase < 0)?(2*M_PI + phase):phase)*TOF_ANGLE_TO_PHASE_FACTOR;
676  t->_amplitude[index] = std::abs(c);
677 
678  t->_ambient[index] = 0;

682

685 
686

688

692  return false;
693  }
694

696  {
697  logger(LOG_ERROR) << "ToFFrameGenerator: Failed to apply phase offset correction" << std::endl;
698  return false;
699  }
700 
701  t->_histogram.clear(); // No histogram

707 bool ToFFrameGenerator::_generateToFRawIQFrame(const FramePtr &in, FramePtr &out)
708 {
709  if(_frameType == ToF_I_Q && _dataArrangeMode != 0)
710  {
711  logger(LOG_ERROR) << "ToFFrameGenerator: Data arrange mode is expected to be zero. It is " << _dataArrangeMode << std::endl;
712  return false;
713  }
714 
715  RawDataFramePtr rawDataFrame = std::dynamic_pointer_cast<RawDataFrame>(in);
716 
717  if(!rawDataFrame)
718  {
719  logger(LOG_ERROR) << "ToFFrameGenerator: Input data frame is not of raw data type." << std::endl;
720  return false;
721  }
722 
723  ToFRawIQFrameTemplate<int16_t> *t;
724 
725  t = dynamic_cast<ToFRawIQFrameTemplate<int16_t> *>(out.get());
726 
727  if(!t)
728  {
729  t = new ToFRawIQFrameTemplate<int16_t>();
730  out = FramePtr(t);
731  }
732 
733  t->size = _size;
734  t->id = rawDataFrame->id;
735  t->timestamp = rawDataFrame->timestamp;
736 
737  uint sizeFactor = 2;
738 
739  if(_frameType == ToF_QUAD)
740  sizeFactor = _quadCount;
741 
742  if(rawDataFrame->data.size() < _size.height*_size.width*sizeFactor)
743  {
744  logger(LOG_ERROR) << "ToFFrameGenerator: Incomplete raw data size = " << rawDataFrame->data.size() << ". Required size = " << _size.height*_size.width*2 << std::endl;
745  return false;
746  }
747 
748  t->_i.resize(_size.width*_size.height);
749  t->_q.resize(_size.width*_size.height);
750 
751  auto index1 = 0, index2 = 0;
752 
753  uint16_t *data = (uint16_t *)rawDataFrame->data.data();
754 
755  if(_frameType == ToF_I_Q)
756  {
757  for (auto i = 0; i < _size.height; i++)
758  {
759  for (auto j = 0; j < _size.width; j++)
760  {
761  index1 = i*_size.width*2 + j*2;
762  index2 = i*_size.width + j;
763 
764  t->_i[index2] = (int16_t)data[index1];
765  t->_q[index2] = (int16_t)data[index1 + 1];
766  }
767  }
768  }
769  else if(_frameType == ToF_QUAD)
770  {
771  uint16_t min = 0, max = 65535;
772  for (auto i = 0; i < _size.height; i++)
773  {
774  for (auto j = 0; j < _size.width; j++)
775  {
776  index2 = i*_size.width + j;
777  index1 = index2*_quadCount;
778 
779  double ii = 0, iq = 0;
780  int16_t d;
781 
782  for(auto k = 0; k < _quadCount; k++)
783  {
784  min = std::min(data[index1 + k], min);
785  max = std::max(data[index1 + k], max);
786  d = (((int16_t)(data[index1 + k] << 4)) >> 4); // Sign extension...
787  ii += d*_cosineTable[k];
788  iq += d*_sineTable[k];
789  }
790 
791  t->_i[index2] = (int16_t)ii;
792  t->_q[index2] = (int16_t)iq;
793  }
794  }
795 
796  logger(LOG_INFO) << "min = " << min << ", max = " << max << std::endl;
797  }
798 
799  return true;
800 }
