164 bool FrameStreamReader::_init()
165 {
166  if(!_stream.good())
167  return false;
168 
169  _currentFrameIndex = _currentPacketIndex = 0;
170 
171  _stream.read(_header.version, 2);
172  _stream.read((char *)&_header.generatorIDs, sizeof(GeneratorIDType)*3);
173 
174  if(_stream.fail() | _stream.bad())
175  {
176  logger(LOG_ERROR) << "FrameStreamReader: Failed to read frame headers." << std::endl;
177  return false;
178  }
179 
180  if(!_sys.getFrameGenerator(DepthCamera::FRAME_RAW_FRAME_PROCESSED, _header.generatorIDs[0], _frameGenerator[0]) ||
181  !_sys.getFrameGenerator(DepthCamera::FRAME_DEPTH_FRAME, _header.generatorIDs[1], _frameGenerator[1]) ||
182  !_sys.getFrameGenerator(DepthCamera::FRAME_XYZI_POINT_CLOUD_FRAME, _header.generatorIDs[2], _frameGenerator[2]))
183  {
184  logger(LOG_ERROR) << "FrameStreamReader: Failed to get necessary frame generators to read stream." << std::endl;
185  return false;
186  }
187 
188  DepthFrameGeneratorPtr p = std::dynamic_pointer_cast<DepthFrameGenerator>(_frameGenerator[1]);
189 
190  if(!p || !p->setProcessedFrameGenerator(_frameGenerator[0]))
191  {
192  logger(LOG_ERROR) << "FrameStreamReader: Could not initialize depth frame generator" << std::endl;
193  return false;
194  }
195 
196  _allPacketOffsets.reserve(500);
197  _dataPacketLocation.reserve(500);
198  _configPacketLocation.reserve(500);
199 
200  while(!_stream.eof())
201  {
202  uint offset = _stream.tellg();
203  if(!_dataPacket.readHeader(_stream))
204  break;
205 
206  if(!_dataPacket.verifyMagic())
207  {
208  logger(LOG_ERROR) << "FrameStreamReader: Got invalid packet at offset = " << _stream.tellg() << std::endl;
209  continue;
210  }
211 
212  if(_dataPacket.type == FrameStreamPacket::PACKET_DATA)
213  {
214  _allPacketOffsets.push_back(offset);
215  _dataPacketLocation.push_back(_allPacketOffsets.size() - 1);
216  _stream.seekg(_dataPacket.size, std::ios::cur);
217  }
218  else if(_dataPacket.type == FrameStreamPacket::PACKET_GENERATOR_CONFIG)
219  {
220  _allPacketOffsets.push_back(offset);
221  _configPacketLocation.push_back(_allPacketOffsets.size() - 1);
222  _stream.seekg(_dataPacket.size, std::ios::cur);
223  }
224  else
225  {
226  logger(LOG_ERROR) << "FrameStreamReader: Got invalid packet type = " << _dataPacket.type << std::endl;
227  }
228  }
229 
230  _stream.clear();
231  if(_allPacketOffsets.size())
232  _stream.seekg(_allPacketOffsets[0], std::ios::beg);
233 
234  logger(LOG_DEBUG) << "stream state: fail = " << _stream.fail() << ", bad = " << _stream.bad() << ", eof = " << _stream.eof() << std::endl;
235 
236  readNext(); // Read initial frame parameters
237  seekTo(0);
238 
239  return true;
240 }
