374 template <typename T>
375 bool ConfigurationFile::_getData(const String &fileName, Vector<T> &data)
376 {
377  bool loadFromFile = false;
378  if(_dataFiles.find(fileName) == _dataFiles.end())
379  {
380  if(_mainConfigurationFile && _parentID >= 0)
381  {
382  ConfigurationFile *parentConfig = _mainConfigurationFile->getCameraProfile(_parentID);
383 
384  // TODO This does not handle circular references between profiles
385  if(parentConfig && parentConfig->_getData<T>(fileName, data))
386  return true;
387  else
388  loadFromFile = true;
389  }
390  else
391  loadFromFile = true;
392  }
393 
394  if(loadFromFile)
395  {
396  Configuration c;
397 
398  String f = fileName;
399  if(!c.getConfFile(f))
400  {
401  logger(LOG_ERROR) << "ConfigurationFile: Could not locate file '" << fileName << "'" << std::endl;
402  return false;
403  }
404 
405  InputFileStream fs(f, std::ios::in | std::ios::binary | std::ios::ate);
406 
407  if(!fs.good())
408  {
409  logger(LOG_ERROR) << "ConfigurationFile: Could not open file '" << fileName << "'" << std::endl;
410  return false;
411  }
412 
413  int size = fs.tellg();
414 
415  if(size == 0)
416  {
417  logger(LOG_ERROR) << "ConfigurationFile: Null config data '" << f << "'" << std::endl;
418  return false;
419  }
420 
421  Vector<ByteType> &d = _dataFiles[fileName];
422 
423  d.resize(size);
424  fs.seekg(std::ios::beg);
425  fs.clear();
426  fs.read((char *)d.data(), size);
427  }
428 
429  Vector<ByteType> &d = _dataFiles[fileName];
430 
431  data.resize((d.size() + sizeof(T)/2)/sizeof(T));
432 
433  memcpy(data.data(), d.data(), d.size());
434 
435  return true;
436 }
