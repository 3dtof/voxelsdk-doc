257 template <typename T>
258 bool ConfigurationFile::_getData(const String &fileName, Vector<T> &data)
259 {
260  if(_dataFiles.find(fileName) == _dataFiles.end())
261  {
262  Configuration c;
263 
264  String f = fileName;
265  if(!c.getConfFile(f))
266  {
267  logger(LOG_ERROR) << "ConfigurationFile: Could not locate file '" << fileName << "'" << std::endl;
268  return false;
269  }
270 
271  InputFileStream fs(f, std::ios::in | std::ios::binary | std::ios::ate);
272 
273  if(!fs.good())
274  {
275  logger(LOG_ERROR) << "ConfigurationFile: Could not open file '" << fileName << "'" << std::endl;
276  return false;
277  }
278 
279  int size = fs.tellg();
280 
281  if(size == 0)
282  {
283  logger(LOG_ERROR) << "ConfigurationFile: Null config data '" << f << "'" << std::endl;
284  return false;
285  }
286 
287  Vector<ByteType> &d = _dataFiles[fileName];
288 
289  d.resize(size);
290  fs.seekg(std::ios::beg);
291  fs.clear();
292  fs.read((char *)d.data(), size);
293  }
294 
295  Vector<ByteType> &d = _dataFiles[fileName];
296 
297  data.resize((d.size() + sizeof(T)/2)/sizeof(T));
298 
299  memcpy(data.data(), d.data(), d.size());
300 
301  return true;
302 }
