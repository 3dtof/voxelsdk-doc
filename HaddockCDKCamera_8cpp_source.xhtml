188 bool HaddockCDKCamera::_setStreamerFrameSize(const FrameSize &s)
189 {
190  USBBulkStreamer *streamer = dynamic_cast<USBBulkStreamer *>(&*_streamer);
191 
192  if(!streamer)

198  int bytesPerPixel;

204  logger(LOG_ERROR) << "HaddockCDKCamera: Could not get current bytes per pixel" << std::endl;
205  return false;

213  if(!streamer->setBufferSize(s.height*s.width*scaleFactor))
214  {

216  return false;
217  }
218 
219  return true;

237 bool HaddockCDKCamera::_getMaximumFrameRate(FrameRate &frameRate, const FrameSize &forFrameSize) const
238 {
239  int opClockFrequency, bytesPerPixel;
240 
241  if(!_get(OP_CLK_FREQ, opClockFrequency) || !_get(PIXEL_DATA_SIZE, bytesPerPixel))
242  {
243  logger(LOG_ERROR) << "HaddockCDKCamera: Could not get " << OP_CLK_FREQ << " or " << PIXEL_DATA_SIZE << std::endl;
244  return false;
245  }
246 
247  opClockFrequency = 24/(1 << opClockFrequency);
248 
249  uint numerator = opClockFrequency*1000000,
250  denominator = bytesPerPixel*forFrameSize.width*forFrameSize.height;
251 
252  uint g = gcd(numerator, denominator);
253 
254  frameRate.numerator = 0.8*numerator/g; // 90% of maximum
255  frameRate.denominator = denominator/g;
256 
257  return true;
258 }
