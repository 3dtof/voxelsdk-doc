357  if(p != _parameters.end())

360 
361  if(param == 0)

364  return false;

367  if(!param->get(value, refresh))

370  return false;

372 
373  return true;

377  logger(LOG_ERROR) << "DepthCamera: Unknown parameter " << _id << "." << name << std::endl;
