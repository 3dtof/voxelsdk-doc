118 bool VoxelDCamera::_init()
119 {
120  USBDevice &d = (USBDevice &)*_device;
121 
122  DevicePtr controlDevice;
123 
124  if(d.productID() == VOXEL_D_PRODUCT_ID1)
125  {
126  Vector<DevicePtr> devices = DeviceScanner::scan();
127 
128  uint repeatCount = 0;
129  for(auto &d1: devices)
130  {
131  if(d1->interfaceID() != Device::USB)
132  continue;
133 
134  USBDevice &usbd = (USBDevice &)*d1;
135 
136  if(usbd.vendorID() == VOXEL_D_VENDOR_ID && usbd.productID() == VOXEL_D_PRODUCT_ID2 &&
137  (d.serialNumber().size() == 0 || d.serialNumber() == usbd.serialNumber()))
138  repeatCount++;
139  }
140 
141 
142  _downloader = Ptr<Downloader>(new USBDownloader(_device));
143  if(!_downloader->download(configFile.get("core", "fw"))) // TODO: This needs to come from a configuration
144  {
145  logger(LOG_ERROR) << "VoxelDCamera: Firmware download failed" << std::endl;
146  return false;
147  }
148  else
149  std::this_thread::sleep_for(std::chrono::milliseconds(2000)); // wait for device to get ready after loading firmware
150 
151  controlDevice = DevicePtr(new USBDevice(VOXEL_D_VENDOR_ID, VOXEL_D_PRODUCT_ID2, _device->serialNumber(),
152  _device->channelID(), _device->description(), _device->serialIndex(), repeatCount > 0));
153  _device = controlDevice;
154  _makeID();
155  }
156  else
157  controlDevice = _device;
158 
159  _programmer = Ptr<RegisterProgrammer>(new VoxelXUProgrammer(
160  { {0x2D, 1}, {0x58, 3}, {0x5C, 3} },
161  controlDevice));
162  _streamer = Ptr<Streamer>(new UVCStreamer(controlDevice));
163 
164  if(!_programmer->isInitialized() || !_streamer->isInitialized())
165  return false;
166 
167  if(!_addParameters({
168  ParameterPtr(new VoxelDMixVoltageParameter(*_programmer)),
169  ParameterPtr(new VoxelDIlluminationVoltageParameter(*_programmer)),
170  ParameterPtr(new VoxelDIlluminationPolarityParameter(*this, *_programmer))
171  }))
172  {
173  return false;
174  }
175 
176  if(!ToFTintinCamera::_init())
177  return false;
178 
179 
180  return true;
181 }

200 bool VoxelDCamera::_setStreamerFrameSize(const FrameSize &s)
201 {
202  UVCStreamer *streamer = dynamic_cast<UVCStreamer *>(&*_streamer);
203 
204  if(!streamer)

214 
215  if(!_get(PIXEL_DATA_SIZE, bytesPerPixel))

218  return false;
219  }
220 
221  if(bytesPerPixel == 4)

223 
224  if(!_getFrameRate(m.frameRate))

227  return false;

230  if(!streamer->setVideoMode(m))

232  logger(LOG_ERROR) << "VoxelDCamera: Could not set video mode for UVC" << std::endl;
233  return false;
234  }
235
