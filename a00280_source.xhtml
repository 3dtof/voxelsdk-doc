78 ParameterPtr ParameterDMLParser::_getParameter(TinyXML2::XMLElement *property, String &id, bool &skipIfNull)
79 {
80  TinyXML2::XMLElement *range, *value;
81 
82  skipIfNull = false;
83 
84  const char *s;
85 
86  if(!(s = property->Attribute("id")))
87  {
88  logger(LOG_ERROR) << "Found a 'property' with no 'id'. " << property->GetText() << std::endl;
89  return 0;
90  }
91 
92  id = s;
93 
94  if(id == "693")
95  std::cout << std::endl;
96 
97  uint bitCount = property->UnsignedAttribute("totalBits"),
98  min = property->UnsignedAttribute("min"),
99  max = property->UnsignedAttribute("max");
100 
101  if(!(range = _goTo(property, { "rangeList", "range" }, false)))
102  {
103  if(bitCount != 0)
104  logger(LOG_ERROR) << "ParameterDMLParser: Could not get 'range' for parameter with id = '" << id << "' with bitCount = " << bitCount << std::endl;
105  skipIfNull = true;
106  return 0; // No range => no valid parameter
107  }
108 
109  uint address = range->UnsignedAttribute("address"),
110  msb = range->UnsignedAttribute("msb"),
111  lsb = range->UnsignedAttribute("lsb"),
112  bankId = range->UnsignedAttribute("bankId");
113 
114  if(address >= 256)
115  {
116  logger(LOG_ERROR) << "ParameterDMLParser: Address value = '" << address << "' which is beyond 256 for parameter with id = '" << id << "'" << std::endl;
117  return 0;
118  }
119 
120  address = (bankId << 8) + address;
121 
122  if(msb < lsb)
123  {
124  logger(LOG_ERROR) << "ParameterDMLParser: Found a parameter with id ='" << id << "' which has msb (" << msb << ") < lsb(" << lsb << ")" << std::endl;
125  return 0;
126  }
127 
128  bool strobe = property->BoolAttribute("strobe"),
129  sign = property->BoolAttribute("signed"),
130  readOnly = property->BoolAttribute("readonly");
131 
132 
133  Parameter::IOType ioType = readOnly?Parameter::IO_READ_ONLY:Parameter::IO_READ_WRITE;
134 
135  String units, description;
136 
137  if((s = property->Attribute("units")))
138  units = s;
139 
140  if((s = property->Attribute("desc")))
141  description = s;
142 
143  if(!(value = _goTo(property, { "valueList", "value" }, false))) // No value list present? => not enum
144  {
145  if(bitCount == 1) // This boolean does not have value descriptions :(
146  {
147  logger(LOG_DEBUG) << "ParameterDMLParser: Found a boolean parameter with id = '" << id << "' which does not have valueList" << std::endl;
148 
149  bool defaultValue = property->BoolAttribute("default");
150 
151  if(!strobe)
152  {
153  logger(LOG_DEBUG) << "ParameterDMLParser: Adding boolean parameter with id = '" << id << "'" << std::endl;
154  return ParameterPtr(new BoolParameter(_registerProgrammer, "", address, _wordLength, lsb, {"", ""}, {"", ""}, defaultValue, "", description,
155  ioType));
156  }
157  else
158  {
159  logger(LOG_DEBUG) << "ParameterDMLParser: Adding strobe (boolean) parameter with id = '" << id << "'" << std::endl;
160  return ParameterPtr(new StrobeBoolParameter(_registerProgrammer, "", address, _wordLength, lsb, {"", ""}, {"", ""}, defaultValue, "", description,
161  ioType));
162  }
163  }
164  else if(bitCount > 1)
165  {
166  if(strobe)
167  logger(LOG_WARNING) << "ParameterDMLParser: Found a non-boolean parameter with id ='" << id << "' marked as strobe. Ignoring 'strobe' for this." << std::endl;
168 
169  if(sign)
170  {
171  if((int)max < (int)min)
172  {
173  logger(LOG_ERROR) << "ParameterDMLParser: Found a parameter with id ='" << id << "' which has max (" << max << ") < min(" << min << ")" << std::endl;
174  return 0;
175  }
176 
177  logger(LOG_DEBUG) << "ParameterDMLParser: Adding int parameter with id = '" << id << "'" << std::endl;
178  int defaultValue = property->IntAttribute("default");
179  return ParameterPtr(new IntegerParameter(_registerProgrammer, "", units, address, _wordLength, msb, lsb, min, max, defaultValue, "", description,
180  ioType));
181  }
182  else
183  {
184  if(max < min)
185  {
186  logger(LOG_ERROR) << "ParameterDMLParser: Found a parameter with id ='" << id << "' which has max (" << max << ") < min(" << min << ")" << std::endl;
187  return 0;
188  }
189 
190  logger(LOG_DEBUG) << "ParameterDMLParser: Adding unsigned int parameter with id = '" << id << "'" << std::endl;
191  uint defaultValue = property->UnsignedAttribute("default");
192  return ParameterPtr(new UnsignedIntegerParameter(_registerProgrammer, "", units, address, _wordLength, msb, lsb, min, max, defaultValue, "", description,
193  ioType));
194  }
195  }
196  else //bitCount == 0
197  {
198  logger(LOG_WARNING) << "ParameterDMLParser: Found a parameter with id ='" << id << "' which has bitCount = 0." << std::endl;
199  skipIfNull = true;
200  return 0;
201  }
202  }
203  else // value list is present => enum
204  {
205  if(bitCount == 1) // boolean
206  {
207  bool defaultValue = property->BoolAttribute("default");
208 
209  Vector<String> valueMeaning, valueDescription;
210 
211  valueMeaning.resize(2);
212  valueDescription.resize(2);
213 
214  for(; value; value = value->NextSiblingElement())
215  {
216  String m, d;
217 
218  if((s = value->Attribute("meaning")))
219  m = s;
220 
221  if((s = value->Attribute("desc")))
222  d = s;
223 
224  int v = value->IntAttribute("value");
225 
226  if(v == 0 || v == 1)
227  {
228  valueMeaning[v] = m;
229  valueDescription[v] = d;
230  }
231  else
232  {
233  logger(LOG_ERROR) << "ParameterDMLParser: Found a boolean parameter with id = '" << id << "' which has possible value as '" << v << "'" << std::endl;
234  return 0;
235  }
236  }
237 
238  if(!strobe)
239  {
240  logger(LOG_DEBUG) << "ParameterDMLParser: Adding boolean parameter with id = '" << id << "'" << std::endl;
241  return ParameterPtr(new BoolParameter(_registerProgrammer, "", address, _wordLength, lsb, valueDescription, valueMeaning, defaultValue, "", description,
242  ioType));
243  }
244  else
245  {
246  logger(LOG_DEBUG) << "ParameterDMLParser: Adding strobe (boolean) parameter with id = '" << id << "'" << std::endl;
247  return ParameterPtr(new StrobeBoolParameter(_registerProgrammer, "", address, _wordLength, lsb, valueDescription, valueMeaning, defaultValue, "", description,
248  ioType));
249  }
250  }
251  else if(bitCount > 1)
252  {
253  if(strobe)
254  logger(LOG_WARNING) << "ParameterDMLParser: Found a non-boolean parameter with id ='" << id << "' marked as strobe. Ignoring 'strobe' for this." << std::endl;
255 
256  Vector<String> valueMeaning, valueDescription;
257  Vector<int> values;
258 
259  for(; value; value = value->NextSiblingElement())
260  {
261  String m, d;
262 
263  if((s = value->Attribute("meaning")))
264  m = s;
265 
266  if((s = value->Attribute("desc")))
267  d = s;
268 
269  valueMeaning.push_back(m);
270  valueDescription.push_back(d);
271  values.push_back(value->IntAttribute("value"));
272  }
273 
274  logger(LOG_DEBUG) << "ParameterDMLParser: Adding enum parameter with id = '" << id << "'" << std::endl;
275 
276  int defaultValue = property->IntAttribute("default");
277  return ParameterPtr(new EnumParameter(_registerProgrammer, "", address, _wordLength, msb, lsb, values, valueDescription, valueMeaning, defaultValue, "", description,
278  ioType));
279  }
280  else //bitCount == 0
281  {
282  logger(LOG_ERROR) << "ParameterDMLParser: Found a parameter with id ='" << id << "' which has bitCount = 0 but has value list." << std::endl;
283  return 0;
284  }

290 bool ParameterDMLParser::getParameters(Vector<ParameterPtr> &parameters)
291 {
292  if(!isInitialized())
293  return false;
294 
295  if(!_prepare())
296  return false;
297 
298  Map<String, ParameterPtr> paramMap; // id -> paramter
299 
300  String id;
301 
302  TinyXML2::XMLElement *x = _propertyList->FirstChildElement("property");
303 
304  ParameterPtr p;
305 
306  bool skipIfNull = false;
307 
308  for(; x; x = x->NextSiblingElement())
309  {
310  if(!(p = _getParameter(x, id, skipIfNull)) && !skipIfNull)
311  return false;
312 
313  if(!p)
314  continue;
315 
316  if(paramMap.find(id) != paramMap.end())
317  {
318  logger(LOG_ERROR) << "ParameterDMLParser: Found an existing parameter with id = '" << id << "'" << std::endl;
319  return false;
320  }
321 
322  paramMap[id] = p;
323  }
324 
325  logger(LOG_DEBUG) << "ParameterDMLParser: Total number of valid parameters = " << paramMap.size() << std::endl;
326 
327  TinyXML2::XMLElement *section = _sectionList->FirstChildElement("section");
328 
329  String name;
330 
331  for(; section; section = section->NextSiblingElement())
332  {
333  if(!section->Attribute("name"))
334  {
335  logger(LOG_ERROR) << "ParameterDMLParser: Found a section with no name in DML file '" << _xmlFileName << std::endl;
336  return false;
337  }
338

340 
341  for(; group; group = group->NextSiblingElement())
342  {
343  if(!group->Attribute("name"))
344  {
345  logger(LOG_ERROR) << "ParameterDMLParser: Found a group with no name in section " << section->Attribute("name") << " in DML file '" << _xmlFileName << std::endl;
346  return false;
347  }
348

350 
351  for(; property; property = property->NextSiblingElement())
352  {

356  return false;
357  }
358 
359  name = property->Attribute("name");
360  id = property->Attribute("propertyId");
361 
362  auto p = paramMap.find(id);
363 
364  if(p != paramMap.end())
365  {
366  p->second->setName(name);
367  parameters.push_back(p->second);
368  }
369  else if(name != "...")
370  {
371  logger(LOG_ERROR) << "ParameterDMLParser: Could not find a parameter with id = " << id << ", in " << section->Attribute("name") << "." << group->Attribute("name") << std::endl;
372  return false;
373  }
374  }
375  }
376  }
377 
378  logger(LOG_DEBUG) << "ParameterDMLParser: Total number of valid parameters = " << parameters.size() << std::endl;
379 
380  return true;
381 }
