75 Point PointCloudTransform::_screenToNormalizedScreen(const Point &screen, bool verify)
76 {
77  int iters = 100;
78  float xs, ys;
79  float yss = ys = (screen.y - cy) / fy;
80  float xss = xs = (screen.x - cx) / fx;
81 
82  for(int j = 0; j < iters; j++)
83  {
84  float r2 = xs * xs + ys * ys;
85  float icdist = 1.0f / (1 + ((k3 * r2 + k2) * r2 + k1) * r2);
86  float deltaX = 2 * p1 * xs * ys + p2 * (r2 + 2 * xs * xs);
87  float deltaY = p1 * (r2 + 2 * ys * ys) + 2 * p2 * xs * ys;
88  xs = (xss - deltaX)*icdist;
89  ys = (yss - deltaY)*icdist;
90  }
91 
92  if(verify)
93  {
94  float x_ = xs;
95  float y_ = ys;
96  float r2 = x_ * x_ + y_ * y_;

99 
100  float x__ = x_ * (1.0f + k1 * r2 + k2 * r4 + k3 * r6) + 2.0f * p1 * x_ * y_ + p2 * (r2 + 2.0f * x_ * x_);
101  float y__ = y_ * (1.0f + k1 * r2 + k2 * r4 + k3 * r6) + p1 * (r2 + 2.0f * y_ * y_) + 2.0f * p2 * x_ * y_;
102 
103  if(fabs(x__ - xss) > FLOAT_EPSILON || fabs(y__ - yss) > FLOAT_EPSILON)

318 {
319  //test if point is lying in view frustrum

321  float r = p.dot(rightClippingNormal);

323  float t = p.dot(topClippingNormal);

326

328  float y_ = p.y / p.z;
329 
330  float r2 = x_ * x_ + y_ * y_;
331  float r4 = r2 * r2;
332  float r6 = r2 * r4;

335  float yss = y_ * (1.0f + k1 * r2 + k2 * r4 + k3 * r6) + p1 * (r2 + 2.0f * y_ * y_) + 2.0f * p2 * x_ * y_;
