92  buffer.append( reinterpret_cast<char*>( &descSizeConv ), sizeof( descSizeConv ) );
93  type = convertor( type );
94  buffer.append( reinterpret_cast<char*>( &type ), sizeof( type ) );
95  buffer.append( name );
96  buffer.append( 1, '\x00' );
97  const char pad[] = { '\0', '\0', '\0', '\0' };
98  if ( nameLen % sizeof( Elf_Word ) != 0 ) {
99  buffer.append( pad, sizeof( Elf_Word ) -
100  nameLen % sizeof( Elf_Word ) );
101  }
102  if ( desc != 0 && descSize != 0 ) {
103  buffer.append( reinterpret_cast<const char*>( desc ), descSize );
104  if ( descSize % sizeof( Elf_Word ) != 0 ) {
105  buffer.append( pad, sizeof( Elf_Word ) -
106  descSize % sizeof( Elf_Word ) );
107  }
108  }
109 
110  note_start_positions.push_back( note_section->get_size() );
111  note_section->append_data( buffer );
112  }

129 
130  while ( current + 3*sizeof( Elf_Word ) <= size ) {
131  note_start_positions.push_back( current );
132  Elf_Word namesz = convertor(
133  *(Elf_Word*)( data + current ) );
134  Elf_Word descsz = convertor(
135  *(Elf_Word*)( data + current + sizeof( namesz ) ) );
136 
137  int align = sizeof( Elf_Xword );
138  if ( elf_file.get_class() == ELFCLASS32 ) {
139  align = sizeof( Elf_Word );
140  }
141 
142  current += 3*sizeof( Elf_Word ) +
143  ( ( namesz + align - 1 ) / align ) * align +
144  ( ( descsz + align - 1 ) / align ) * align;
145  }
146  }
