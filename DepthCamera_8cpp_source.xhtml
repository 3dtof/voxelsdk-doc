119 void DepthCamera::_captureLoop()
120 {
121  uint consecutiveCaptureFails = 0;
122 
123  while(_running)
124  {
125  uint32_t callBackTypesToBeCalled = _callBackTypesRegistered;
126 
127  if(consecutiveCaptureFails > 100)
128  {
129  logger(LOG_ERROR) << "DepthCamera: 100 consecutive failures in capture of frame. Stopping stream for " << id() << std::endl;
130  _running = false;
131  continue;
132  }
133 
134  if((_callBackTypesRegistered == 0 || _callBackTypesRegistered == FRAME_RAW_FRAME_UNPROCESSED) && !isSavingFrameStream()) // Only unprocessed frame types requested or none requested?
135  {
136  auto f = _rawFrameBuffers.get();
137 
138  if(!_captureRawUnprocessedFrame(*f))
139  {
140  consecutiveCaptureFails++;
141  continue;
142  }
143 
144  if(_callback[FRAME_RAW_FRAME_UNPROCESSED])
145  {
146  FilterSet<RawFrame>::FrameSequence _frameBuffers;
147  _frameBuffers.push_front(f);
148 
149  if(!_unprocessedFilters.applyFilter(_frameBuffers))
150  {
151  logger(LOG_ERROR) << "DepthCamera: Failed to apply filters on raw unprocessed frame" << std::endl;
152  consecutiveCaptureFails++;
153  continue;
154  }
155 
156  _callback[FRAME_RAW_FRAME_UNPROCESSED](*this, (Frame &)(**_frameBuffers.begin()), FRAME_RAW_FRAME_UNPROCESSED);
157 
158  _writeToFrameStream(**_frameBuffers.begin());
159  }
160  else
161  {
162  _writeToFrameStream(*f);
163  }
164 
165  consecutiveCaptureFails = 0;
166  }
167  else
168  {
169  auto f1 = _rawFrameBuffers.get();
170  if(!_captureRawUnprocessedFrame(*f1))
171  {
172  consecutiveCaptureFails++;
173  continue;
174  }
175 
176  FilterSet<RawFrame>::FrameSequence _unprocessedFrameBuffers;
177  _unprocessedFrameBuffers.push_front(f1);
178 
179  if(!_unprocessedFilters.applyFilter(_unprocessedFrameBuffers))
180  {
181  logger(LOG_ERROR) << "DepthCamera: Failed to apply filters on raw unprocessed frame" << std::endl;
182  consecutiveCaptureFails++;
183  continue;
184  }
185 
186  if(!_callbackAndContinue(callBackTypesToBeCalled, FRAME_RAW_FRAME_UNPROCESSED, ***_unprocessedFrameBuffers.begin()) && !isSavingFrameStream())
187  {
188  consecutiveCaptureFails = 0;
189  continue;
190  }
191 
192  auto f = _rawFrameBuffers.get();
193 
194  if(!_processRawFrame(**_unprocessedFrameBuffers.begin(), *f))
195  {
196  consecutiveCaptureFails++;
197  continue;
198  }
199 
200  FilterSet<RawFrame>::FrameSequence _processedFrameBuffers;
201  _processedFrameBuffers.push_front(f);
202 
203  if(!_processedFilters.applyFilter(_processedFrameBuffers))
204  {
205  logger(LOG_ERROR) << "DepthCamera: Failed to apply filters on raw processed frame" << std::endl;
206  consecutiveCaptureFails++;
207  continue;
208  }
209 
210  if(!_callbackAndContinue(callBackTypesToBeCalled, FRAME_RAW_FRAME_PROCESSED, ***_processedFrameBuffers.begin()) && !isSavingFrameStream())
211  {
212  consecutiveCaptureFails = 0;
213  continue;
214  }
215 
216  auto d = _depthFrameBuffers.get();
217 
218  if(!_convertToDepthFrame(**_processedFrameBuffers.begin(), *d))
219  {
220  consecutiveCaptureFails++;
221  continue;
222  }
223 
224  FilterSet<DepthFrame>::FrameSequence _depthFrameBuffers;
225  _depthFrameBuffers.push_front(d);
226 
227  if(!_depthFilters.applyFilter(_depthFrameBuffers))
228  {
229  logger(LOG_ERROR) << "DepthCamera: Failed to apply filters on depth frame" << std::endl;
230  consecutiveCaptureFails++;
231  continue;
232  }
233 
234  if(!_callbackAndContinue(callBackTypesToBeCalled, FRAME_DEPTH_FRAME, ***_depthFrameBuffers.begin()) && !isSavingFrameStream())
235  {
236  consecutiveCaptureFails = 0;
237  continue;
238  }
239 
240  auto p = _pointCloudBuffers.get();
241 
242  if(!_convertToPointCloudFrame(**_depthFrameBuffers.begin(), *p))
243  {
244  consecutiveCaptureFails++;
245  continue;
246  }
247 
248  _callbackAndContinue(callBackTypesToBeCalled, FRAME_XYZI_POINT_CLOUD_FRAME, **p);
249  consecutiveCaptureFails = 0;
250 
251  _writeToFrameStream(**_unprocessedFrameBuffers.begin());
252  }
253  }
254 
255  if(!_running)
256  {
257  closeFrameStream();
258  _stop();
259  }
260 
261  logger(LOG_INFO) << "DepthCamera: Streaming stopped." << std::endl;
262 }

614 bool DepthCamera::_applyConfigParams(const ConfigSet *params)
615 {
616  for(auto i = 0; i < params->paramNames.size(); i++)
617  {
618  if(params->paramNames[i].compare(0, 2, "0x") == 0)
619  {
620  logger(LOG_INFO) << "DepthCamera: Setting register '" << params->paramNames[i] << "'" << std::endl;
621 
622  char *endptr;
623  uint32_t reg = (uint32_t)strtol(params->paramNames[i].c_str(), &endptr, 16);
624  uint32_t value = (uint32_t)strtol(params->get(params->paramNames[i]).c_str(), &endptr, 0);
625 
626  if(!_programmer->writeRegister(reg, value))
627  {
628  logger(LOG_ERROR) << "Failed to write to register @0x" << std::hex << reg << " = 0x" << value << std::dec << std::endl;
629  }
630  continue;
631  }
632  else if(params->paramNames[i] == "frame_rate")
633  {
634  float rate = params->getFloat(params->paramNames[i]);
635  FrameRate r;
636  r.numerator = rate*10000;
637  r.denominator = 10000;
638 
639  uint g = gcd(r.numerator, r.denominator);
640 
641  r.numerator /= g;
642  r.denominator /= g;
643 
644  if(!setFrameRate(r))
645  {
646  logger(LOG_ERROR) << "DepthCamera: Failed to set frame rate to " << rate << "fps" << std::endl;
647  return false;
648  }
649  continue;
650  }
651 
652  logger(LOG_INFO) << "DepthCamera: Setting parameter '" << params->paramNames[i] << "'" << std::endl;
653 
654  const Parameter *p = getParam(params->paramNames[i]).get();
655 
656  if(!p)
657  {
658  logger(LOG_ERROR) << "DepthCamera: Ignoring unknown parameter " << params->paramNames[i] << std::endl;
659  return false;
660  }
661 
662  const BoolParameter *bp = dynamic_cast<const BoolParameter *>(p);
663  const EnumParameter *ep = dynamic_cast<const EnumParameter *>(p);
664  const IntegerParameter *ip = dynamic_cast<const IntegerParameter *>(p);
665  const UnsignedIntegerParameter *up = dynamic_cast<const UnsignedIntegerParameter *>(p);
666  const FloatParameter *fp = dynamic_cast<const FloatParameter *>(p);
667 
668  if(bp)
669  {
670  if(!set(params->paramNames[i], params->getBoolean(params->paramNames[i])))
671  return false;
672  }
673  else if(ip || ep)
674  {
675  if(!set(params->paramNames[i], params->getInteger(params->paramNames[i])))
676  return false;
677  }
678  else if(up)
679  {
680  if(!set(params->paramNames[i], (uint)params->getInteger(params->paramNames[i])))
681  return false;
682  }
683  else if(fp)
684  {
685  if(!set(params->paramNames[i], params->getFloat(params->paramNames[i])))
686  return false;
687  }
688  else
689  {
690  logger(LOG_ERROR) << "DepthCamera: Parameter type unknown for " << params->paramNames[i] << std::endl;
691  return false;
692  }
693  }
694  return true;
695 }
