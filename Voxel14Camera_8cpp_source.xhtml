84 bool Voxel14Camera::_init()
85 {
86  USBDevice &d = (USBDevice &)*_device;
87 
88  DevicePtr controlDevice;
89 
90  if(d.productID() == VOXEL_14_PRODUCT_ID1)
91  {
92  Vector<DevicePtr> devices = DeviceScanner::scan();
93 
94  uint repeatCount = 0;
95  for(auto &d1: devices)
96  {
97  if(d1->interfaceID() != Device::USB)
98  continue;
99 
100  USBDevice &usbd = (USBDevice &)*d1;
101 
102  if(usbd.vendorID() == VOXEL_14_VENDOR_ID && usbd.productID() == VOXEL_14_PRODUCT_ID2 &&
103  (d.serialNumber().size() == 0 || d.serialNumber() == usbd.serialNumber()))
104  repeatCount++;
105  }
106 
107 
108  _downloader = Ptr<Downloader>(new USBDownloader(_device));
109  if(!_downloader->download(configFile.get("core", "fw")))
110  {
111  logger(LOG_ERROR) << "Voxel14Camera: Firmware download failed" << std::endl;
112  return false;
113  }
114  else
115  std::this_thread::sleep_for(std::chrono::milliseconds(2000)); // wait for device to get ready after loading firmware
116 
117  controlDevice = DevicePtr(new USBDevice(VOXEL_14_VENDOR_ID, VOXEL_14_PRODUCT_ID2, _device->serialNumber(),
118  _device->channelID(), _device->description(), _device->serialIndex(), repeatCount > 0));
119  _device = controlDevice;
120  _makeID();
121  }
122  else
123  controlDevice = _device;
124 
125  _programmer = Ptr<RegisterProgrammer>(new VoxelXUProgrammer(
126  { {0x2D, 1}, {0x58, 3}, {0x5C, 3} },
127  controlDevice));
128  _streamer = Ptr<Streamer>(new UVCStreamer(controlDevice));
129 
130  if(!_programmer->isInitialized() || !_streamer->isInitialized())
131  return false;
132 
133  if(!_addParameters({
134  ParameterPtr(new Voxel14MixVoltageParameter(*_programmer)),
135  ParameterPtr(new Voxel14IlluminationVoltageParameter(*_programmer)),
136  }))
137  {
138  return false;
139  }
140 
141  if(!ToFHaddockCamera::_init())
142  return false;
143 
144  {
145  CalibrationInformation &calibInfo = _getCalibrationInformationStructure()[ToF_CALIB_SECT_COMMON_PHASE_OFFSET];
146  Vector<String> params = {ILLUM_VOLTAGE};
147  calibInfo.definingParameters.insert(calibInfo.definingParameters.end(), params.begin(), params.end());
148  }

152  Vector<String> params = {ILLUM_VOLTAGE};

154  }
