175 template <typename FrameType>
176 bool FilterSet<FrameType>::applyFilter(FrameSequence &seq)
177 {
178  Lock<Mutex> _(_accessMutex);
179 
180  if(seq.size() < 1)
181  {
182  logger(LOG_ERROR) << "Require atleast one frame in the sequence, to apply filters" << std::endl;
183  return false;
184  }
185 
186  for(auto f: *this)
187  {
188  auto g = _frameBufferManager.get();
189 
190  FramePtr p = std::dynamic_pointer_cast<Frame>(*g);
191 
192  if(!f->filter(std::dynamic_pointer_cast<Frame>(**seq.begin()), p))
193  {
194  logger(LOG_ERROR) << "FilterSet: Could not apply filter '" << f->name() << "'" << std::endl;
195  return false;
196  }
197 
198  *g = std::dynamic_pointer_cast<FrameType>(p);
199 
200  if(!*g)
201  {
202  logger(LOG_ERROR) << "FilterSet: Got an invalid frame from filter '" << f->name() << "'" << std::endl;
203  return false;
204  }
205 
206  seq.push_front(g);
207  }
208  return true;
