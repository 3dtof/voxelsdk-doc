598 bool TintinCDKCamera::_getMaximumFrameRate(FrameRate &frameRate, const FrameSize &forFrameSize) const
599 {

601 
602  if(!_get(OP_CLK_FREQ, opClockFrequency) || !_get(PIXEL_DATA_SIZE, bytesPerPixel))
603  {
604  logger(LOG_ERROR) << "TintinCDKCamera: Could not get " << OP_CLK_FREQ << " or " << PIXEL_DATA_SIZE << std::endl;
605  return false;
606  }
607 
608  opClockFrequency = 24/(1 << opClockFrequency);
609 
610  uint numerator = opClockFrequency*1000000,
611  denominator = bytesPerPixel*forFrameSize.width*forFrameSize.height;
612 
613  uint g = gcd(numerator, denominator);
614 
615  frameRate.numerator = 0.8*numerator/g; // 90% of maximum
616  frameRate.denominator = denominator/g;

648 bool TintinCDKCamera::_readConfigFromHardware(MainConfigurationFile::ConfigSerialNumberType &serialNumber, TimeStampType &knownTimestamp, SerializedObject &so)
649 {

651

654  Timer t;
655  TimeStampType d = t.getCurentRealTime();

658 
659  uint8_t data[9 + 2 + sizeof(TimeStampType)];
660

662 
663  if(!usbIO->controlTransfer(USBIO::FROM_DEVICE, USBIO::REQUEST_VENDOR, USBIO::RECIPIENT_DEVICE, REQUEST_EEPROM_DATA, 0, 0, data, l))
664  {
665  logger(LOG_ERROR) << "TintinCDKCamera: Failed to read config size in EEPROM." << std::endl;
666  return false;
667  }
668

670  {
671  logger(LOG_ERROR) << "TintinCDKCamera: Invalid config data in EEPROM." << std::endl;
672  return false;
673  }
674 
675  // Ignoring the version info for now
676 
677  //std::cout << "Timestamp in hardware = " << *(TimeStampType *)&data[7] << ", expected = " << knownTimestamp << std::endl;

680  {
681  so.resize(0);
682  return true;
683  }
684 
685  int32_t length = *(uint32_t *)&data[7 + sizeof(TimeStampType)];
686 
687  uint32_t totalSize;
688 
689  if(!_getEEPROMSize(totalSize))
690  return false;
691 
692  if(length > totalSize)
693  {

695  << length << " > " << totalSize << ")" << std::endl;
696  return false;
697  }
698 
699  so.resize(length);
700 
701  uint32_t offset = l, localOffset = 0;
702 
703  const uint32_t maxTransferLength = 4096;
704 
705  while(length > 0)
706  {
707  l = length;
708 
709  if(l > maxTransferLength) l = maxTransferLength;
710 
711  //std::cout << "l = " << l << std::endl;
712 
713  uint16_t offsetLower = (offset & 0xFFFF);
714  uint16_t offsetUpper = (offset >> 16);
715 
716  if(!usbIO->controlTransfer(USBIO::FROM_DEVICE, USBIO::REQUEST_VENDOR, USBIO::RECIPIENT_DEVICE,
717  REQUEST_EEPROM_DATA, offsetUpper, offsetLower, (uint8_t *)so.getBytes().data() + localOffset, l) || l == 0)
718  {
719  logger(LOG_ERROR) << "TintinCDKCamera: Failed to read config data from EEPROM." << std::endl;
720  return false;
721  }
722 
723  length -= l;
724  offset += l;
725  localOffset += l;
726  }
727 
728  logger(LOG_INFO) << "TintinCDKCamera: Received " << so.size() << " bytes from EEPROM in " << (t.getCurentRealTime() - d)*1E-6 << " s" << std::endl;
