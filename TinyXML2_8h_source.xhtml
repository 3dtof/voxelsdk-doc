377  virtual void* Alloc() {
378  if ( !_root ) {
379  // Need a new block.
380  Block* block = new Block();
381  _blockPtrs.Push( block );
382 
383  for( int i=0; i<COUNT-1; ++i ) {
384  block->chunk[i].next = &block->chunk[i+1];
385  }
386  block->chunk[COUNT-1].next = 0;
387  _root = block->chunk;
388  }
389  void* result = _root;
390  _root = _root->next;
391 
392  ++_currentAllocs;
393  if ( _currentAllocs > _maxAllocs ) {
394  _maxAllocs = _currentAllocs;
395  }
